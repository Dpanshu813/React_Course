// title
title E-Learning Platform Data Model

// define tables
user [icon: user, color: yellow]{
  id bigint pk
  name varchar(100)
  role enum('student','instructor')
  email varchar(255)
  password varchar(255)
  created_at datetime
  updated_at datetime
}

course [icon: book, color: blue]{
  id bigint pk
  title varchar(255)
  description varchar(1000)
  content_url varchar(500)
  instructor_id bigint
  created_at datetime
  updated_at datetime
}

enrollment [icon: check-circle, color: green]{
  id bigint pk
  student_id bigint
  course_id bigint
  progress decimal(5,2)
  enrolled_at datetime
  updated_at datetime
}

assessment [icon: file-text, color: orange]{
  id bigint pk
  course_id bigint
  type enum('quiz','assignment')
  max_score int
  title varchar(255)
  description varchar(1000)
  created_at datetime
  updated_at datetime
}

submission [icon: edit-3, color: purple]{
  id bigint pk
  assessment_id bigint
  student_id bigint
  score decimal(5,2)
  submitted_at datetime
  graded_at datetime
  feedback varchar(1000)
}

notification [icon: bell, color: red]{
  id bigint pk
  user_id bigint
  message varchar(1000)
  type enum('deadline','course_update','announcement','submission')
  is_read boolean
  created_at datetime
}

// define relationships
course.instructor_id > user.id
enrollment.student_id > user.id
enrollment.course_id > course.id
assessment.course_id > course.id
submission.assessment_id > assessment.id
submission.student_id > user.id
notification.user_id > user.id
